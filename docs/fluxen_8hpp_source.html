137/* On-disk entry header (6 bytes on disk) */
138struct EntryHeader {
139 uint8_t flags;
140 uint8_t key_len;
141 uint32_t val_len;
142};
143
144/* Serialize header into 6-byte buffer */
145inline void encode_header(uint8_t out[HEADER_SIZE],
146 const EntryHeader &h) noexcept {
147 out[0] = h.flags;
148 out[1] = h.key_len;
149 out[2] = static_cast<uint8_t>(h.val_len & 0xFFu);
150 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
151 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
152 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
153}
154
155/* Deserialise header from a 6-byte buffer */
156inline auto decode_header(const uint8_t in[HEADER_SIZE]) noexcept
157 -> EntryHeader {
158 return {
159 .flags = in[0],
160 .key_len = in[1],
161 .val_len = static_cast<uint32_t>(in[2]) |
162 static_cast<uint32_t>(in[3]) << 8 |
163 static_cast<uint32_t>(in[4]) << 16 |
164 static_cast<uint32_t>(in[5]) << 24,
165 };
166}

244 auto remap() -> bool {

284 auto append(const void *data, size_t len) -> bool {
285 if (len == 0) {

705 template <typename T = std::string>
706 auto get(std::string_view key) const -> std::optional<T> {
707 std::shared_lock lock(mu_);
708 ensure_mapped();
709
710 auto it = index_.find(key);
711 if (it == index_.end()) {
712 return std::nullopt;
713 }
714
715 const auto &entry = it->second;
716
717 if constexpr (std::is_same_v<T, std::string>) {
718 const auto *ptr = file_.ptr() + entry.val_offset;
719 return std::string(reinterpret_cast<const char *>(ptr), entry.val_len);
720 } else {
721 static_assert(std::is_trivially_copyable_v<T>,
722 "fluxen: T must be trivially copyable");
723 if (entry.val_len != static_cast<uint32_t>(sizeof(T))) {
724 return std::nullopt;
725 }
726 T result;
727 std::memcpy(&result, file_.ptr() + entry.val_offset, sizeof(T));
728 return result;
729 }

950
951 std::vector<uint8_t> buf;

953
954 buf.insert(buf.end(), detail::MAGIC, detail::MAGIC + sizeof(detail::MAGIC));
955
956 detail::IndexMap new_index;
957 for (const auto &[key, entry] : index_) {
958 detail::EntryHeader hdr{
959 .flags = detail::FLAG_LIVE,
960 .key_len = static_cast<uint8_t>(key.size()),
961 .val_len = entry.val_len,
962 };
963

965 detail::encode_header(raw, hdr);
966
967 size_t val_off = buf.size() + detail::HEADER_SIZE + key.size();
968
969 buf.insert(buf.end(), raw, raw + detail::HEADER_SIZE);
970 buf.insert(buf.end(), key.begin(), key.end());
971
972 const auto *val_ptr = file_.ptr() + entry.val_offset;
973 buf.insert(buf.end(), val_ptr, val_ptr + entry.val_len);

976 }

1027 void load_index() {
1028 if (file_.size() < sizeof(detail::MAGIC)) {
1029 throw std::runtime_error("fluxen: file too small to be valid");
1030 }
1031
1032 if (std::memcmp(file_.ptr(), detail::MAGIC, sizeof(detail::MAGIC)) != 0) {
1033 throw std::runtime_error(
1034 "fluxen: bad magic. File was not created by fluxen");
1035 }
1036
1037 size_t pos = sizeof(detail::MAGIC);
1038
1039 while (pos + detail::HEADER_SIZE <= file_.size()) {
1040 uint8_t raw[detail::HEADER_SIZE];
1041 std::memcpy(raw, file_.ptr() + pos, detail::HEADER_SIZE);
1042 detail::EntryHeader hdr = detail::decode_header(raw);
1043 pos += detail::HEADER_SIZE;
1044
1045 if (pos + hdr.key_len + hdr.val_len > file_.size()) {
1046 break;
1047 }
1048
1049 std::string key(reinterpret_cast<const char *>(file_.ptr() + pos),
1050 hdr.key_len);
1051 pos += hdr.key_len;
1052
1053 if (hdr.flags == detail::FLAG_TOMB) {
1054 index_.erase(key);
1055 } else {
1056 index_[key] = {.val_offset = pos, .val_len = hdr.val_len};
1057 }

1059 pos += hdr.val_len;

1072 void append_entry(std::string_view key, const uint8_t *val, uint32_t val_len,
1073 bool tombstone) {
1074 assert(key.size() <= detail::MAX_KEY &&
1075 "fluxen: key exceeds 255-byte limit");
1076
1077 detail::EntryHeader hdr{
1078 .flags = tombstone ? detail::FLAG_TOMB : detail::FLAG_LIVE,
1079 .key_len = static_cast<uint8_t>(key.size()),
1080 .val_len = val_len,
1081 };
1082
1083 std::vector<uint8_t> buf;
1084 buf.resize(detail::HEADER_SIZE + key.size() + val_len);
1085 detail::encode_header(buf.data(), hdr);
1086 std::memcpy(buf.data() + detail::HEADER_SIZE, key.data(), key.size());
1087 if (val && val_len) {
1088 std::memcpy(buf.data() + detail::HEADER_SIZE + key.size(), val, val_len);
1089 }
1090
1091 file_.append(buf.data(), buf.size());
1092
1093 if (tombstone) {
1094 if (auto it = index_.find(key); it != index_.end()) {
1095 index_.erase(it);
1096 }
1097 } else {
1098 index_[std::string(key)] = {.val_offset = file_.size() - val_len,
1099 .val_len = val_len};
1100 }
