139/* On-disk entry header (6 bytes on disk) */
140struct EntryHeader {
141 uint8_t flags;
142 uint8_t key_len;
143 uint32_t val_len;
144};
145
146/* Serialize header into 6-byte buffer */
147inline void encode_header(uint8_t out[HEADER_SIZE],
148 const EntryHeader &h) noexcept {
149 out[0] = h.flags;
150 out[1] = h.key_len;
151 out[2] = static_cast<uint8_t>(h.val_len & 0xFFu);
152 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
153 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
154 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
155}
156
157/* Deserialise header from a 6-byte buffer */
158inline auto decode_header(const uint8_t in[HEADER_SIZE]) noexcept
159 -> EntryHeader {
160 return {
161 .flags = in[0],
162 .key_len = in[1],
163 .val_len = static_cast<uint32_t>(in[2]) |
164 static_cast<uint32_t>(in[3]) << 8 |
165 static_cast<uint32_t>(in[4]) << 16 |
166 static_cast<uint32_t>(in[5]) << 24,
167 };
168}

1015 void transaction(const std::function<TxResult(Tx &)> &fn) {
1016 check_poisoned();

1018 TxResult result = fn(tx);
1019 if (result == rollback) {
1020 return;
1021 }
1022
1023 std::vector<uint8_t> batch;
1024 batch.reserve(tx.ops_.size() * 64);
1025 for (const auto &op : tx.ops_) {
1026 detail::EntryHeader hdr{
1027 .flags = op.is_delete ? detail::FLAG_TOMB : detail::FLAG_LIVE,
1028 .key_len = static_cast<uint8_t>(op.key.size()),
1029 .val_len = op.is_delete ? 0u : static_cast<uint32_t>(op.val.size()),
1030 };
1031 uint8_t raw[detail::HEADER_SIZE];
1032 detail::encode_header(raw, hdr);
1033 batch.insert(batch.end(), raw, raw + detail::HEADER_SIZE);
1034 batch.insert(batch.end(), op.key.begin(), op.key.end());
1035 if (!op.is_delete) {
1036 batch.insert(batch.end(), op.val.begin(), op.val.end());
1037 }
1038 }
1039
1040 std::unique_lock lock(mu_);
1041
1042 const size_t size_before = file_.size();
1043
1044 if (!file_.append(batch.data(), batch.size())) {
1045 throw std::runtime_error("fluxen: transaction append failed");
1046 }
1047
1048 if (!file_.sync()) {
1049 if (!file_.truncate(size_before)) {
1050 poisoned_ = true;
1051 throw std::runtime_error(
1052 "fluxen: transaction fsync failed and truncation failed. Database "
1053 "file may contain a partial tail entry");
1054 }
1055 throw std::runtime_error("fluxen: transaction fsync failed");
1056 }
1057
1058 size_t pos = 0;
1059 for (const auto &op : tx.ops_) {
1060 pos += detail::HEADER_SIZE + op.key.size();
1061 if (op.is_delete) {
1062 index_.erase(op.key);
1063 } else {
1064 index_[op.key] = {.val_offset = size_before + pos,
1065 .val_len = static_cast<uint32_t>(op.val.size())};
1066 pos += op.val.size();
1067 }
1068 }

1252 void append_entry(std::string_view key, const uint8_t *val, uint32_t val_len,
1253 bool tombstone) {
1254 if (key.empty() || key.size() > detail::MAX_KEY) {
1255 throw std::runtime_error("fluxen: key must be between 1 and 255 bytes");
1256 }
1257
1258 detail::EntryHeader hdr{
1259 .flags = tombstone ? detail::FLAG_TOMB : detail::FLAG_LIVE,
1260 .key_len = static_cast<uint8_t>(key.size()),
1261 .val_len = val_len,
1262 };
1263
1264 std::vector<uint8_t> buf;
1265 buf.resize(detail::HEADER_SIZE + key.size() + val_len);
1266 detail::encode_header(buf.data(), hdr);
1267 std::memcpy(buf.data() + detail::HEADER_SIZE, key.data(), key.size());
1268 if (val && val_len) {
1269 std::memcpy(buf.data() + detail::HEADER_SIZE + key.size(), val, val_len);
