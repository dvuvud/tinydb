124/* On-disk entry header (6 bytes on disk) */
125struct EntryHeader {
126 uint8_t flags;
127 uint8_t key_len;
128 uint32_t val_len;
129};
130
131/* Serialize header into 6-byte buffer */
132inline void encode_header(uint8_t out[HEADER_SIZE], const EntryHeader& h) noexcept {
133 out[0] = h.flags;
134 out[1] = h.key_len;
135 out[2] = static_cast<uint8_t>( h.val_len & 0xFFu);
136 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
137 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
138 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
139}
140
141/* Deserialise header from a 6-byte buffer */
142inline auto decode_header(const uint8_t in[HEADER_SIZE]) noexcept -> EntryHeader {
143 return {
144 .flags = in[0],
145 .key_len = in[1],
146 .val_len = static_cast<uint32_t>(in[2])
147 | static_cast<uint32_t>(in[3]) << 8
148 | static_cast<uint32_t>(in[4]) << 16
149 | static_cast<uint32_t>(in[5]) << 24,
150 };
151}

203 unmap();

206 file_size_ = size_;

237 if (len == 0) {
238 return true;

240#ifdef _WIN32

270 SetEndOfFile(file_);

565 template <typename T = std::string>
566 auto get(std::string_view key) const -> std::optional<T> {
567 std::scoped_lock lock(mu_);
568 auto it = index_.find(std::string(key));
569 if (it == index_.end()) {
570 return std::nullopt;
571 }
572
573 const auto& entry = it->second;
574
575 if constexpr (std::is_same_v<T, std::string>) {
576 const auto* ptr = file_.ptr() + entry.val_offset;
577 return std::string(reinterpret_cast<const char*>(ptr), entry.val_len);
578 } else {
579 static_assert(std::is_trivially_copyable_v<T>, "tinydb: T must be trivially copyable");
580 if (entry.val_len != static_cast<uint32_t>(sizeof(T))) {
581 return std::nullopt;
582 }
583 T result;
584 std::memcpy(&result, file_.ptr() + entry.val_offset, sizeof(T));
585 return result;

737 void transaction(const std::function<TxResult(Tx&)>& fn) {
738 Tx tx;
739 TxResult result = fn(tx);
740 if (result == rollback) return;
741
742 std::scoped_lock lock(mu_);
743 for (const auto& op : tx.ops_) {
744 append_entry(
745 op.key,
746 op.is_delete ? nullptr : op.val.data(),
747 op.is_delete ? 0 : op.val.size(),
748 op.is_delete
749 );
750 }
751 file_.sync();

786 void compact() {
787 std::scoped_lock lock(mu_);
788
789 std::vector<uint8_t> buf;
790 buf.reserve(file_.size());
791
792 buf.insert(
793 buf.end(),
794 detail::MAGIC,
795 detail::MAGIC + sizeof(detail::MAGIC)
796 );
797
798 std::unordered_map<std::string, detail::IndexEntry> new_index;
799 for (const auto& [key, entry] : index_) {
800 detail::EntryHeader hdr{
801 .flags = detail::FLAG_LIVE,
802 .key_len = static_cast<uint8_t>(key.size()),
803 .val_len = entry.val_len,
804 };
805
806 uint8_t raw[detail::HEADER_SIZE];
807 detail::encode_header(raw, hdr);
808
809 size_t val_off = buf.size() + detail::HEADER_SIZE + key.size();
810
811 buf.insert(buf.end(), raw, raw + detail::HEADER_SIZE);
812 buf.insert(buf.end(), key.begin(), key.end());
813
814 const auto* val_ptr = file_.ptr() + entry.val_offset;
815 buf.insert(buf.end(), val_ptr, val_ptr + entry.val_len);
816
817 new_index[key] = { .val_offset=val_off, .val_len=entry.val_len };
818 }
819
820 if (!file_.rewrite(buf)) {
821 throw std::runtime_error("tinydb: compact failed during rewrite");
822 }
823
824 index_ = std::move(new_index);
825 }

867 void load_index() {
868 if (file_.size() < sizeof(detail::MAGIC)) {
869 throw std::runtime_error("tinydb: file too small to be valid");
870 }
871
872 if (std::memcmp(file_.ptr(), detail::MAGIC, sizeof(detail::MAGIC)) != 0) {
873 throw std::runtime_error("tinydb: bad magic. File was not created by tinydb");
874 }
875
876 size_t pos = sizeof(detail::MAGIC);
877
878 while (pos + detail::HEADER_SIZE <= file_.size()) {
879 uint8_t raw[detail::HEADER_SIZE];
880 std::memcpy(raw, file_.ptr() + pos, detail::HEADER_SIZE);
881 detail::EntryHeader hdr = detail::decode_header(raw);
882 pos += detail::HEADER_SIZE;
883
884 if (pos + hdr.key_len + hdr.val_len > file_.size()) {
885 break;
886 }
887
888 std::string key(
889 reinterpret_cast<const char*>(file_.ptr() + pos),
890 hdr.key_len
891 );
892 pos += hdr.key_len;
893
894 if (hdr.flags == detail::FLAG_TOMB) {
895 index_.erase(key);
896 } else {
897 index_[key] = { .val_offset=pos, .val_len=hdr.val_len };
898 }
899
900 pos += hdr.val_len;
901 }

913 void append_entry(std::string_view key,
914 const uint8_t* val,
915 uint32_t val_len,
916 bool tombstone) {
917 assert(key.size() <= detail::MAX_KEY && "tinydb: key exceeds 255-byte limit");
918
919 detail::EntryHeader hdr{
920 .flags = tombstone ? detail::FLAG_TOMB : detail::FLAG_LIVE,
921 .key_len = static_cast<uint8_t>(key.size()),
922 .val_len = val_len,
923 };
924
925 uint8_t raw[detail::HEADER_SIZE];
926 detail::encode_header(raw, hdr);
927
928 file_.append(raw, detail::HEADER_SIZE);
929 file_.append(key.data(), key.size());
930 if (val && val_len) {
931 file_.append(val, val_len);
932 }
933
934 if (tombstone) {
935 index_.erase(std::string(key));
936 } else {
937 size_t val_off = file_.size() - val_len;
938 index_[std::string(key)] = { .val_offset=val_off, .val_len=val_len };
939 }
940 }
