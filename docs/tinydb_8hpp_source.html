137/* On-disk entry header (6 bytes on disk) */
138struct EntryHeader {
139 uint8_t flags;
140 uint8_t key_len;
141 uint32_t val_len;
142};
143
144/* Serialize header into 6-byte buffer */
145inline void encode_header(uint8_t out[HEADER_SIZE], const EntryHeader& h) noexcept {
146 out[0] = h.flags;
147 out[1] = h.key_len;
148 out[2] = static_cast<uint8_t>( h.val_len & 0xFFu);
149 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
150 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
151 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
152}
153
154/* Deserialise header from a 6-byte buffer */
155inline auto decode_header(const uint8_t in[HEADER_SIZE]) noexcept -> EntryHeader {
156 return {
157 .flags = in[0],
158 .key_len = in[1],
159 .val_len = static_cast<uint32_t>(in[2])
160 | static_cast<uint32_t>(in[3]) << 8
161 | static_cast<uint32_t>(in[4]) << 16
162 | static_cast<uint32_t>(in[5]) << 24,
163 };
164}

188class WritePreferringRWLock {
189public:
190 void lock() {
191 std::unique_lock lk(mu_);
192 ++waiting_writers_;
193 write_cv_.wait(lk, [this] () -> bool{
194 return active_readers_ == 0 && active_writers_ == 0;
195 });
196 --waiting_writers_;
197 ++active_writers_;
198 }
199
200 void unlock() {
201 std::unique_lock lk(mu_);
202 --active_writers_;
203 if (waiting_writers_ > 0) {
204 write_cv_.notify_one();
205 } else {
206 read_cv_.notify_all();
207 }
208 }
209
210 void lock_shared() {
211 std::unique_lock lk(mu_);
212 read_cv_.wait(lk, [this] () -> bool{
213 return waiting_writers_ == 0 && active_writers_ == 0;
214 });
215 ++active_readers_;
216 }
217
218 void unlock_shared() {
219 std::unique_lock lk(mu_);
220 if (--active_readers_ == 0 && waiting_writers_ > 0) {
221 write_cv_.notify_one();
222 }
223 }
224private:
225 std::mutex mu_;
226 std::condition_variable read_cv_;
227 std::condition_variable write_cv_;
228 int active_readers_ = 0;
229 int active_writers_ = 0;

664 template <typename T = std::string>
665 auto get(std::string_view key) const -> std::optional<T> {
666 std::shared_lock lock(mu_);
667 ensure_mapped();
668
669 auto it = index_.find(std::string(key));
670 if (it == index_.end()) {
671 return std::nullopt;
672 }
673
674 const auto& entry = it->second;
675
676 if constexpr (std::is_same_v<T, std::string>) {
677 const auto* ptr = file_.ptr() + entry.val_offset;
678 return std::string(reinterpret_cast<const char*>(ptr), entry.val_len);
679 } else {
680 static_assert(std::is_trivially_copyable_v<T>, "tinydb: T must be trivially copyable");
681 if (entry.val_len != static_cast<uint32_t>(sizeof(T))) {
682 return std::nullopt;
683 }
684 T result;
685 std::memcpy(&result, file_.ptr() + entry.val_offset, sizeof(T));
686 return result;
687 }

899 void compact() {
900 std::scoped_lock lock(mu_);
901
902 std::vector<uint8_t> buf;
903 buf.reserve(file_.size());
904
905 buf.insert(
906 buf.end(),
907 detail::MAGIC,
908 detail::MAGIC + sizeof(detail::MAGIC)
909 );
910
911 std::unordered_map<std::string, detail::IndexEntry> new_index;
912 for (const auto& [key, entry] : index_) {
913 detail::EntryHeader hdr{
914 .flags = detail::FLAG_LIVE,
915 .key_len = static_cast<uint8_t>(key.size()),
916 .val_len = entry.val_len,
917 };
918
919 uint8_t raw[detail::HEADER_SIZE];
920 detail::encode_header(raw, hdr);
921
922 size_t val_off = buf.size() + detail::HEADER_SIZE + key.size();
923
924 buf.insert(buf.end(), raw, raw + detail::HEADER_SIZE);
925 buf.insert(buf.end(), key.begin(), key.end());
926
927 const auto* val_ptr = file_.ptr() + entry.val_offset;
928 buf.insert(buf.end(), val_ptr, val_ptr + entry.val_len);
929
930 new_index[key] = { .val_offset=val_off, .val_len=entry.val_len };
931 }
932
933 if (!file_.rewrite(buf)) {
934 throw std::runtime_error("tinydb: compact failed during rewrite");
935 }
936
937 index_ = std::move(new_index);
938 }

984 void load_index() {
985 if (file_.size() < sizeof(detail::MAGIC)) {
986 throw std::runtime_error("tinydb: file too small to be valid");
987 }
988
989 if (std::memcmp(file_.ptr(), detail::MAGIC, sizeof(detail::MAGIC)) != 0) {
990 throw std::runtime_error("tinydb: bad magic. File was not created by tinydb");
991 }
992
993 size_t pos = sizeof(detail::MAGIC);
994
995 while (pos + detail::HEADER_SIZE <= file_.size()) {
996 uint8_t raw[detail::HEADER_SIZE];
997 std::memcpy(raw, file_.ptr() + pos, detail::HEADER_SIZE);
998 detail::EntryHeader hdr = detail::decode_header(raw);
999 pos += detail::HEADER_SIZE;
1000
1001 if (pos + hdr.key_len + hdr.val_len > file_.size()) {
1002 break;
1003 }
1004
1005 std::string key(
1006 reinterpret_cast<const char*>(file_.ptr() + pos),
1007 hdr.key_len
1008 );
1009 pos += hdr.key_len;
1010
1011 if (hdr.flags == detail::FLAG_TOMB) {
1012 index_.erase(key);
1013 } else {
1014 index_[key] = { .val_offset=pos, .val_len=hdr.val_len };
1015 }
1016
1017 pos += hdr.val_len;
1018 }

1030 void append_entry(std::string_view key,
1031 const uint8_t* val,
1032 uint32_t val_len,
1033 bool tombstone) {
1034 assert(key.size() <= detail::MAX_KEY && "tinydb: key exceeds 255-byte limit");

1037 .flags = tombstone ? detail::FLAG_TOMB : detail::FLAG_LIVE,
1038 .key_len = static_cast<uint8_t>(key.size()),
1039 .val_len = val_len,
1040 };
1041
1042 uint8_t raw[detail::HEADER_SIZE];
1043 detail::encode_header(raw, hdr);
1044
1045 file_.append(raw, detail::HEADER_SIZE);
1046 file_.append(key.data(), key.size());

1048 file_.append(val, val_len);

1052 index_.erase(std::string(key));
1053 } else {
1054 size_t val_off = file_.size() - val_len;
1055 index_[std::string(key)] = { .val_offset=val_off, .val_len=val_len };
