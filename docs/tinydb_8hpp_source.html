120/* On-disk entry header (6 bytes on disk) */
121struct EntryHeader {
122 uint8_t flags;
123 uint8_t key_len;
124 uint32_t val_len;
125};
126
127/* Serialize header into 6-byte buffer */
128inline void encode_header(uint8_t out[HEADER_SIZE], const EntryHeader& h) noexcept {
129 out[0] = h.flags;
130 out[1] = h.key_len;
131 out[2] = static_cast<uint8_t>( h.val_len & 0xFFu);
132 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
133 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
134 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
135}
136
137/* Deserialise header from a 6-byte buffer */
138inline EntryHeader decode_header(const uint8_t in[HEADER_SIZE]) noexcept {
139 return {
140 .flags = in[0],
141 .key_len = in[1],
142 .val_len = static_cast<uint32_t>(in[2])
143 | static_cast<uint32_t>(in[3]) << 8
144 | static_cast<uint32_t>(in[4]) << 16
145 | static_cast<uint32_t>(in[5]) << 24,
146 };
147}

537 template <typename T = std::string>
538 std::optional<T> get(std::string_view key) const {
539 std::lock_guard lock(mu_);
540 auto it = index_.find(std::string(key));
541 if (it == index_.end()) {
542 return std::nullopt;
543 }
544
545 const auto& entry = it->second;
546
547 if constexpr (std::is_same_v<T, std::string>) {
548 const auto* ptr = file_.ptr() + entry.val_offset;
549 return std::string(reinterpret_cast<const char*>(ptr), entry.val_len);
550 } else {
551 static_assert(std::is_trivially_copyable_v<T>, "tinydb: T must be trivially copyable");
552 if (entry.val_len != static_cast<uint32_t>(sizeof(T))) {
553 return std::nullopt;
554 }
555 T result;
556 std::memcpy(&result, file_.ptr() + entry.val_offset, sizeof(T));
557 return result;
558 }

703 void transaction(std::function<TxResult(Tx&)> fn) {
704 Tx tx;
705 TxResult result = fn(tx);
706 if (result == rollback) return;
707
708 std::lock_guard lock(mu_);
709 for (const auto& op : tx.ops_) {
710 append_entry(
711 op.key,
712 op.is_delete ? nullptr : op.val.data(),
713 op.is_delete ? 0 : op.val.size(),
714 op.is_delete
715 );
716 }
717 file_.sync();
718 }

747 void compact() {
748 std::lock_guard lock(mu_);
749
750 std::vector<uint8_t> buf;
751 buf.reserve(file_.size());
752
753 buf.insert(
754 buf.end(),
755 detail::MAGIC,
756 detail::MAGIC + sizeof(detail::MAGIC)
757 );
758
759 std::unordered_map<std::string, detail::IndexEntry> new_index;
760 for (const auto& [key, entry] : index_) {
761 detail::EntryHeader hdr{
762 .flags = detail::FLAG_LIVE,
763 .key_len = static_cast<uint8_t>(key.size()),
764 .val_len = entry.val_len,
765 };
766
767 uint8_t raw[detail::HEADER_SIZE];
768 detail::encode_header(raw, hdr);
769
770 size_t val_off = buf.size() + detail::HEADER_SIZE + key.size();
771
772 buf.insert(buf.end(), raw, raw + detail::HEADER_SIZE);
773 buf.insert(buf.end(), key.begin(), key.end());
774
775 const auto* val_ptr = file_.ptr() + entry.val_offset;
776 buf.insert(buf.end(), val_ptr, val_ptr + entry.val_len);
777
778 new_index[key] = { val_off, entry.val_len };
779 }
780
781 if (!file_.rewrite(buf)) {
782 throw std::runtime_error("tinydb: compact failed during rewrite");
783 }
784
785 index_ = std::move(new_index);
786 }

829 void load_index() {
830 if (file_.size() < sizeof(detail::MAGIC)) {
831 throw std::runtime_error("tinydb: file too small to be valid");
832 }
833
834 if (std::memcmp(file_.ptr(), detail::MAGIC, sizeof(detail::MAGIC)) != 0) {
835 throw std::runtime_error("tinydb: bad magic. File was not created by tinydb");
836 }
837
838 size_t pos = sizeof(detail::MAGIC);
839
840 while (pos + detail::HEADER_SIZE <= file_.size()) {
841 uint8_t raw[detail::HEADER_SIZE];
842 std::memcpy(raw, file_.ptr() + pos, detail::HEADER_SIZE);
843 detail::EntryHeader hdr = detail::decode_header(raw);
844 pos += detail::HEADER_SIZE;
845
846 if (pos + hdr.key_len + hdr.val_len > file_.size()) {
847 break;
848 }
849
850 std::string key(
851 reinterpret_cast<const char*>(file_.ptr() + pos),
852 hdr.key_len
853 );
854 pos += hdr.key_len;
855
856 if (hdr.flags == detail::FLAG_TOMB) {
857 index_.erase(key);
858 } else {
859 index_[key] = { pos, hdr.val_len };
860 }
861
862 pos += hdr.val_len;
863 }
864 }

875 void append_entry(std::string_view key,
876 const uint8_t* val,
877 uint32_t val_len,
878 bool tombstone) {
879 assert(key.size() <= detail::MAX_KEY && "tinydb: key exceeds 255-byte limit");
880
881 detail::EntryHeader hdr{
882 .flags = tombstone ? detail::FLAG_TOMB : detail::FLAG_LIVE,
883 .key_len = static_cast<uint8_t>(key.size()),
884 .val_len = val_len,
885 };
886
887 uint8_t raw[detail::HEADER_SIZE];
888 detail::encode_header(raw, hdr);
889
890 file_.append(raw, detail::HEADER_SIZE);
891 file_.append(key.data(), key.size());
892 if (val && val_len) {
893 file_.append(val, val_len);
894 }
895 file_.remap();
896
897 if (tombstone) {
898 index_.erase(std::string(key));
899 } else {
900 size_t val_off = file_.size() - val_len;
901 index_[std::string(key)] = { val_off, val_len };
902 }
903 }
