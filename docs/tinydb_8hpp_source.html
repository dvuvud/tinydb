136/* On-disk entry header (6 bytes on disk) */
137struct EntryHeader {
138 uint8_t flags;
139 uint8_t key_len;
140 uint32_t val_len;
141};
142
143/* Serialize header into 6-byte buffer */
144inline void encode_header(uint8_t out[HEADER_SIZE], const EntryHeader& h) noexcept {
145 out[0] = h.flags;
146 out[1] = h.key_len;
147 out[2] = static_cast<uint8_t>( h.val_len & 0xFFu);
148 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
149 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
150 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
151}
152
153/* Deserialise header from a 6-byte buffer */
154inline auto decode_header(const uint8_t in[HEADER_SIZE]) noexcept -> EntryHeader {
155 return {
156 .flags = in[0],
157 .key_len = in[1],
158 .val_len = static_cast<uint32_t>(in[2])
159 | static_cast<uint32_t>(in[3]) << 8
160 | static_cast<uint32_t>(in[4]) << 16
161 | static_cast<uint32_t>(in[5]) << 24,
162 };
163}

695 auto get(std::string_view key) const -> std::optional<T> {
696 std::shared_lock lock(mu_);
697 ensure_mapped();
698
699 auto it = index_.find(std::string(key));
700 if (it == index_.end()) {
701 return std::nullopt;
702 }
703
704 const auto& entry = it->second;
705
706 if constexpr (std::is_same_v<T, std::string>) {
707 const auto* ptr = file_.ptr() + entry.val_offset;
708 return std::string(reinterpret_cast<const char*>(ptr), entry.val_len);
709 } else {
710 static_assert(std::is_trivially_copyable_v<T>, "tinydb: T must be trivially copyable");

712 return std::nullopt;
713 }

715 std::memcpy(&result, file_.ptr() + entry.val_offset, sizeof(T));

934 void compact() {
935 std::unique_lock lock(mu_);
936
937 if (file_.is_dirty()) {
938 file_.remap();
939 }
940
941 std::vector<uint8_t> buf;
942 buf.reserve(file_.size());
943
944 buf.insert(
945 buf.end(),
946 detail::MAGIC,
947 detail::MAGIC + sizeof(detail::MAGIC)
948 );
949
950 std::unordered_map<std::string, detail::IndexEntry> new_index;
951 for (const auto& [key, entry] : index_) {
952 detail::EntryHeader hdr{
953 .flags = detail::FLAG_LIVE,
954 .key_len = static_cast<uint8_t>(key.size()),
955 .val_len = entry.val_len,
956 };
957
958 uint8_t raw[detail::HEADER_SIZE];
959 detail::encode_header(raw, hdr);
960
961 size_t val_off = buf.size() + detail::HEADER_SIZE + key.size();
962
963 buf.insert(buf.end(), raw, raw + detail::HEADER_SIZE);
964 buf.insert(buf.end(), key.begin(), key.end());
965
966 const auto* val_ptr = file_.ptr() + entry.val_offset;
967 buf.insert(buf.end(), val_ptr, val_ptr + entry.val_len);
968
969 new_index[key] = { .val_offset=val_off, .val_len=entry.val_len };
970 }
971
972 if (!file_.rewrite(buf)) {
973 throw std::runtime_error("tinydb: compact failed during rewrite");
974 }
975
976 index_ = std::move(new_index);
977 }

1028 if (std::memcmp(file_.ptr(), detail::MAGIC, sizeof(detail::MAGIC)) != 0) {

1069 void append_entry(std::string_view key,
1070 const uint8_t* val,
1071 uint32_t val_len,
1072 bool tombstone) {
1073 assert(key.size() <= detail::MAX_KEY && "tinydb: key exceeds 255-byte limit");
1074
1075 detail::EntryHeader hdr{
1076 .flags = tombstone ? detail::FLAG_TOMB : detail::FLAG_LIVE,
1077 .key_len = static_cast<uint8_t>(key.size()),
1078 .val_len = val_len,
1079 };
1080
1081 uint8_t raw[detail::HEADER_SIZE];
1082 detail::encode_header(raw, hdr);
1083
1084 file_.append(raw, detail::HEADER_SIZE);
1085 file_.append(key.data(), key.size());
1086 if (val && val_len) {
1087 file_.append(val, val_len);
1088 }
1089
1090 if (tombstone) {
1091 index_.erase(std::string(key));
1092 } else {
1093 size_t val_off = file_.size() - val_len;
1094 index_[std::string(key)] = { .val_offset=val_off, .val_len=val_len };
1095 }
