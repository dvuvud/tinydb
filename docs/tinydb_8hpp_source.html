135/* On-disk entry header (6 bytes on disk) */
136struct EntryHeader {
137 uint8_t flags;
138 uint8_t key_len;
139 uint32_t val_len;
140};
141
142/* Serialize header into 6-byte buffer */
143inline void encode_header(uint8_t out[HEADER_SIZE], const EntryHeader& h) noexcept {
144 out[0] = h.flags;
145 out[1] = h.key_len;
146 out[2] = static_cast<uint8_t>( h.val_len & 0xFFu);
147 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
148 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
149 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
150}
151
152/* Deserialise header from a 6-byte buffer */
153inline auto decode_header(const uint8_t in[HEADER_SIZE]) noexcept -> EntryHeader {
154 return {
155 .flags = in[0],
156 .key_len = in[1],
157 .val_len = static_cast<uint32_t>(in[2])
158 | static_cast<uint32_t>(in[3]) << 8
159 | static_cast<uint32_t>(in[4]) << 16
160 | static_cast<uint32_t>(in[5]) << 24,
161 };
162}

251 auto append(const void* data, size_t len) -> bool {
252 if (len == 0) {
253 return true;
254 }
255#ifdef _WIN32
256 DWORD written = 0;
257 SetFilePointer(file_, 0, nullptr, FILE_END);
258 if (!WriteFile(file_, data, static_cast<DWORD>(len), &written, nullptr)
259 || written != static_cast<DWORD>(len)) {
260 return false;
261 }
262#else

264 if (::write(fd_, data, len) != static_cast<ssize_t>(len)) {
265 return false;
266 }
267#endif

605 template <typename T = std::string>
606 auto get(std::string_view key) const -> std::optional<T> {
607 std::shared_lock lock(mu_);
608 ensure_mapped();
609
610 auto it = index_.find(std::string(key));
611 if (it == index_.end()) {
612 return std::nullopt;
613 }
614
615 const auto& entry = it->second;
616
617 if constexpr (std::is_same_v<T, std::string>) {
618 const auto* ptr = file_.ptr() + entry.val_offset;
619 return std::string(reinterpret_cast<const char*>(ptr), entry.val_len);
620 } else {
621 static_assert(std::is_trivially_copyable_v<T>, "tinydb: T must be trivially copyable");
622 if (entry.val_len != static_cast<uint32_t>(sizeof(T))) {
623 return std::nullopt;
624 }
625 T result;
626 std::memcpy(&result, file_.ptr() + entry.val_offset, sizeof(T));
627 return result;

792 Tx tx;

840 void compact() {
841 std::unique_lock lock(mu_);
842
843 if (file_.is_dirty()) {
844 file_.remap();
845 }
846
847 std::vector<uint8_t> buf;
848 buf.reserve(file_.size());
849
850 buf.insert(
851 buf.end(),
852 detail::MAGIC,
853 detail::MAGIC + sizeof(detail::MAGIC)
854 );
855
856 std::unordered_map<std::string, detail::IndexEntry> new_index;
857 for (const auto& [key, entry] : index_) {
858 detail::EntryHeader hdr{
859 .flags = detail::FLAG_LIVE,
860 .key_len = static_cast<uint8_t>(key.size()),
861 .val_len = entry.val_len,
862 };
863
864 uint8_t raw[detail::HEADER_SIZE];
865 detail::encode_header(raw, hdr);
866
867 size_t val_off = buf.size() + detail::HEADER_SIZE + key.size();
868
869 buf.insert(buf.end(), raw, raw + detail::HEADER_SIZE);
870 buf.insert(buf.end(), key.begin(), key.end());
871
872 const auto* val_ptr = file_.ptr() + entry.val_offset;
873 buf.insert(buf.end(), val_ptr, val_ptr + entry.val_len);
874
875 new_index[key] = { .val_offset=val_off, .val_len=entry.val_len };
876 }
877
878 if (!file_.rewrite(buf)) {
879 throw std::runtime_error("tinydb: compact failed during rewrite");
880 }
881
882 index_ = std::move(new_index);
883 }

929 void load_index() {
930 if (file_.size() < sizeof(detail::MAGIC)) {
931 throw std::runtime_error("tinydb: file too small to be valid");
932 }
933
934 if (std::memcmp(file_.ptr(), detail::MAGIC, sizeof(detail::MAGIC)) != 0) {
935 throw std::runtime_error("tinydb: bad magic. File was not created by tinydb");
936 }
937
938 size_t pos = sizeof(detail::MAGIC);
939
940 while (pos + detail::HEADER_SIZE <= file_.size()) {
941 uint8_t raw[detail::HEADER_SIZE];
942 std::memcpy(raw, file_.ptr() + pos, detail::HEADER_SIZE);
943 detail::EntryHeader hdr = detail::decode_header(raw);
944 pos += detail::HEADER_SIZE;
945
946 if (pos + hdr.key_len + hdr.val_len > file_.size()) {
947 break;
948 }
949
950 std::string key(
951 reinterpret_cast<const char*>(file_.ptr() + pos),
952 hdr.key_len
953 );
954 pos += hdr.key_len;
955
956 if (hdr.flags == detail::FLAG_TOMB) {
957 index_.erase(key);
958 } else {
959 index_[key] = { .val_offset=pos, .val_len=hdr.val_len };
960 }
961
962 pos += hdr.val_len;
963 }

975 void append_entry(std::string_view key,
976 const uint8_t* val,
977 uint32_t val_len,
978 bool tombstone) {
979 assert(key.size() <= detail::MAX_KEY && "tinydb: key exceeds 255-byte limit");
980
981 detail::EntryHeader hdr{
982 .flags = tombstone ? detail::FLAG_TOMB : detail::FLAG_LIVE,
983 .key_len = static_cast<uint8_t>(key.size()),
984 .val_len = val_len,
985 };
986
987 uint8_t raw[detail::HEADER_SIZE];
988 detail::encode_header(raw, hdr);
989
990 file_.append(raw, detail::HEADER_SIZE);
991 file_.append(key.data(), key.size());
992 if (val && val_len) {
993 file_.append(val, val_len);
994 }
995
996 if (tombstone) {
997 index_.erase(std::string(key));
998 } else {
999 size_t val_off = file_.size() - val_len;
1000 index_[std::string(key)] = { .val_offset=val_off, .val_len=val_len };
1001 }
1002 }
